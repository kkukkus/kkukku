// 터치 제스처 추가 (스와이프로 페이지 넘기기) let touchStartX = 0; let touchStartY = 0; viewer.addEventListener('touchstart', e => { touchStartX = e.touches[0].clientX; touchStartY = e.touches[0].clientY; }); viewer.addEventListener('touchend', e => { const touchEndX = e.changedTouches[0].clientX; const touchEndY = e.changedTouches[0].clientY; const deltaX = touchEndX - touchStartX; const deltaY = touchEndY - touchStartY; // 가로 스와이프가 세로 스와이프보다 클 때만 페이지 넘기기 if (Math.abs(deltaX) > Math.abs(deltaY) && Math.abs(deltaX) > 50) { if (deltaX > 0) { // 오른쪽 스와이프 - 이전 페이지 prevBtn.click(); } else { // 왼쪽 스와이프 - 다음 페이지 nextBtn.click(); } } }); // 스크롤로 페이지 넘기기 (PC: 마우스 휠, 모바일: 터치 스크롤) let scrollTimeout = null; let lastScrollTime = 0; const scrollThrottle = 800; // 스크롤 간격 (ms) viewer.addEventListener('wheel', e => { const now = Date.now(); if (now - lastScrollTime < scrollThrottle) return; // 스크롤 방향 감지 if (Math.abs(e.deltaY) > Math.abs(e.deltaX)) { e.preventDefault(); if (e.deltaY > 0) { // 아래로 스크롤 - 다음 페이지 nextBtn.click(); } else { // 위로 스크롤 - 이전 페이지 prevBtn.click(); } lastScrollTime = now; } }, { passive: false }); // 모바일 터치 스크롤 감지 let scrollStartY = 0; let isScrolling = false; viewer.addEventListener('touchstart', e => { if (e.touches.length === 1) { scrollStartY = e.touches[0].clientY; isScrolling = false; } }, { passive: true }); viewer.addEventListener('touchmove', e => { if (e.touches.length === 1 && !isScrolling) { const deltaY = e.touches[0].clientY - scrollStartY; // 세로 스크롤 감지 (가로 스와이프와 구분) if (Math.abs(deltaY) > 30) { const now = Date.now(); if (now - lastScrollTime < scrollThrottle) return; isScrolling = true; if (deltaY < 0) { // 위로 스크롤 - 다음 페이지 nextBtn.click(); } else { // 아래로 스크롤 - 이전 페이지 prevBtn.click(); } lastScrollTime = now; } } }, { passive: true });
